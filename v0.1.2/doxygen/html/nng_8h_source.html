61template<>
62struct fmt::formatter<nng::msg> : formatter<std::string>
63{
64 auto format(const nng::msg& m, format_context& ctx) const
65 -> decltype(ctx.out())
66 {
67 if (m.body().size() < sizeof(std::size_t))
68 {
69 return fmt::format_to(
70 ctx.out(), "size:{}={}+{}, hash:<error>, pipe:0x{:X}",
71 m.header().size() + m.body().size(), m.header().size(), m.body().size(),
72 m.get_pipe().id());
73 }
74 else
75 {
76 std::size_t h = pars::hash_from_msg(m);
77
78 return fmt::format_to(
79 ctx.out(), "size:{}={}+{}, hash:0x{:X}, pipe:0x{:X}",
80 m.header().size() + m.body().size(), m.header().size(), m.body().size(),
81 h, m.get_pipe().id());
82 }
83 }
84};
