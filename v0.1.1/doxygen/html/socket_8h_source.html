55static cmode cmode_from_string(const char* str)
56{
57 auto str_view = std::string_view(str);
58
59 if (str_view.compare("dial") == 0)
60 return cmode::dial;
61 else if (str_view.compare("listen") == 0)
62 return cmode::listen;
63
64 throw new std::runtime_error(fmt::format("Unable to parse {} to CMODE", str));
65}

163private:
164 void pipe_cb(nng_pipe cp, nng_pipe_ev ev)
165 {
166 auto pv = nng::pipe_view{cp};
167
168 switch (ev)
169 {
170 case NNG_PIPE_EV_ADD_PRE: {
171 pars::debug(SL, lf::net, "Pipe 0x{:X} creating! [{}]", pv.id(), *this);
172
173 router_m.queue_fire(ev::creating_pipe{}, id(), *this, net::pipe{pv});
174 }
175 break;
176
177 case NNG_PIPE_EV_ADD_POST: {
178 pars::debug(SL, lf::net, "Pipe 0x{:X} created! [{}]", pv.id(), *this);
179
180 router_m.queue_fire(ev::pipe_created{}, id(), *this, net::pipe{pv});
181 }
182 break;
183
184 case NNG_PIPE_EV_REM_POST: {
185 pars::debug(SL, lf::net, "Pipe 0x{:X} removed! [{}]", pv.id(), *this);
186
187 router_m.queue_fire(ev::pipe_removed{}, id(), *this, net::pipe{pv});
188 }
189 break;
190
191 case NNG_PIPE_EV_NUM: {
192 pars::debug(SL, lf::net, "Pipe 0x{:X} num notified", pv.id());
193 }
194 break;
195 }
196 }

198 void register_pipe_notify()
199 {
200 static auto pipe_cb = [](nng_pipe p, nng_pipe_ev ev, void* self) {
201 static_cast<socket*>(self)->pipe_cb(p, ev);
202 };
203
204 // NOTE: pass this, cant move socket
205 socket_m.pipe_notify(nng::pipe_ev::add_pre, pipe_cb, this);
206 socket_m.pipe_notify(nng::pipe_ev::add_post, pipe_cb, this);
207 socket_m.pipe_notify(nng::pipe_ev::rem_post, pipe_cb, this);
208 socket_m.pipe_notify(nng::pipe_ev::num, pipe_cb, this);
209 }
