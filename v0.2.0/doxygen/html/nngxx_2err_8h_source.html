42enum class err : int;

44[[nodiscard]] inline static const std::error_category& error_category() noexcept
45{
46 static struct : std::error_category
47 {
48 virtual const char* name() const noexcept override { return "nngxx"; }
49
50 virtual std::string message(int e) const override
51 {
52 return nng_strerror(e);
53 }
54 } error_category;
55
56 return error_category;
57}
58
59[[nodiscard]] inline std::error_code make_error_code(err e) noexcept
60{
61 return std::error_code(static_cast<int>(e), error_category());
62}
