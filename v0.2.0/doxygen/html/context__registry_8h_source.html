44class context_registry

46public:
47 context_registry(ev::enqueuer& r, net::socket& s)

53 void stop_all()
54 {
55 for (auto& c : ctx_map_m)
56 c.second.stop();
57 }
58
59 context& emplace()
60 {
61 auto ctx = sock_m.make_ctx();
62
63 auto id = ctx.id();
64
65 auto res = ctx_map_m.try_emplace(id, router_m, std::move(ctx), sock_m);
66
67 if (!res.second)
68 throw std::runtime_error("Unable to emplace a context");
69
70 return res.first->second;
71 }
72
73 context& of(const net::tool_view& t)
74 {
75 if (t.type() != typeid(nngxx::ctx_view))
76 throw std::runtime_error("We need a context here");
77
78 if (ctx_map_m.find(t.id()) == ctx_map_m.end())
79 throw std::runtime_error(fmt::format("Unknown context {}", t.id()));
80
81 return ctx_map_m.at(t.id());
82 }
83
84 void start_recv(int num_ctxs)
85 {
86 for (int i = 0; i < num_ctxs; ++i)
87 emplace().recv();
88 }
89
90private:
91 ev::enqueuer& router_m;
92 socket& sock_m;
93 std::unordered_map<int, context> ctx_map_m;
