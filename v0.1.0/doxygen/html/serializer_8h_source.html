48struct serialize
49{
50 template<event_c event_t>
51 static nng::msg to_network(event_t& ev)
52 {
53 // 1. serialize the event to a stringstream
54 auto ostring = std::ostringstream();
55 auto ostream = std::ostream(ostring.rdbuf());
56 auto ar = cereal::BinaryOutputArchive(ostream);
57 klass<event_t>::serialize(ev, ar);
58
59 // 2. compute the event_hash
60 auto event_hash = uuid<klass<event_t>>::hash;
61
62 // 3. create the nng::msg to hold the hash+event
63 auto serialization = ostring.rdbuf()->view();
64 auto m = nng::make_msg(sizeof(event_hash) + serialization.size());
65 auto b = m.body();
66
67 // 4. append the event hash
68 memcpy(b.data<char>(), &event_hash, sizeof(event_hash));
69
70 // 5. append the serialized event
71 memcpy(b.data<char>() + sizeof(event_hash), serialization.data(),
72 serialization.size());
73
74 pars::debug(SL, lf::event, "Serialized Event [{}] to Message [{}]", ev, m);
75
76 return m;
77 }
78
79 template<event_c event_t>
80 static event_t to_event(const nng::msg& m)
81 {
82 // 1. compute received and requested event hash
83 auto recv_event_hash = hash_from_msg(m);
84 auto req_event_hash = uuid<klass<event_t>>::hash;
85
86 // 2. check they correspond
87 if (recv_event_hash - req_event_hash != 0)
88 throw new std::runtime_error("Requested event mismatch!");
89
90 // 3. remove event hash
91 m.body().trim_u64();
92
93 // 4. deserialize event
94 auto body = m.body();
95 auto view = std::string_view(body.data<char>(), body.size());
96 auto istring = std::ispanstream(view);
97 auto istream = std::istream(istring.rdbuf());
98 auto ar = cereal::BinaryInputArchive(istream);
99
100 auto ev = event_t{};
101 klass<event_t>::serialize(ev, ar);
102
103 pars::debug(SL, lf::event, "Serialized Message [{}] to Event [{}]", m, ev);
104
105 return ev;
106 }
107};
