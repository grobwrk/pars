108template<template<typename> typename kind_of, event_c event_t>
109 requires kind_c<kind_of>
110static std::size_t compute_spec_hash(const kind_of<event_t>& ke)
111{
112 if constexpr (std::is_same_v<kind_of<event_t>, received<nngxx::msg>>)
113 return ke.msg_hash();
114 else
115 return spec<kind_of<event_t>>::hash;
116}

118template<template<typename> typename kind_of, event_c event_t>
119 requires kind_c<kind_of>
120static job make_job(std::size_t j_id, kind_of<event_t> ke)
121{
122 auto h = compute_spec_hash(ke);
