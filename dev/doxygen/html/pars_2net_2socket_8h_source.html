48enum class cmode
49{
50 dial,
51 listen
52};
53
54static cmode cmode_from_string(const char* str)
55{
56 auto str_view = std::string_view(str);
57
58 if (str_view.compare("dial") == 0)
59 return cmode::dial;
60 else if (str_view.compare("listen") == 0)
61 return cmode::listen;
62
63 throw std::runtime_error(fmt::format("Unable to parse {} to CMODE", str));
64}

158private:
159 void pipe_cb(nng_pipe cp, nng_pipe_ev ev)
160 {
161 auto pv = nngxx::pipe_view{cp};
162
163 switch (ev)
164 {
165 case NNG_PIPE_EV_ADD_PRE: {
166 pars::debug(SL, lf::net, "Pipe 0x{:X} creating! [{}]", pv.id(), *this);
167
168 router_m.queue_fire(ev::creating_pipe{}, id(), *this, net::pipe{pv});
169 }
170 break;
171
172 case NNG_PIPE_EV_ADD_POST: {
173 pars::debug(SL, lf::net, "Pipe 0x{:X} created! [{}]", pv.id(), *this);
174
175 router_m.queue_fire(ev::pipe_created{}, id(), *this, net::pipe{pv});
176 }
177 break;
178
179 case NNG_PIPE_EV_REM_POST: {
180 pars::debug(SL, lf::net, "Pipe 0x{:X} removed! [{}]", pv.id(), *this);
181
182 router_m.queue_fire(ev::pipe_removed{}, id(), *this, net::pipe{pv});
183 }
184 break;
185
186 case NNG_PIPE_EV_NUM: {
187 pars::debug(SL, lf::net, "Pipe 0x{:X} num notified", pv.id());
188 }
189 break;
190 }
191 }

193 void register_pipe_notify()
194 {
195 static auto pipe_cb = [](nng_pipe p, nng_pipe_ev ev, void* self) {
196 static_cast<socket*>(self)->pipe_cb(p, ev);
197 };
198
199 // NOTE: pass this, cant move socket
200 socket_m.pipe_notify(nngxx::pipe_ev::add_pre, pipe_cb, this).or_abort();
201 socket_m.pipe_notify(nngxx::pipe_ev::add_post, pipe_cb, this).or_abort();
202 socket_m.pipe_notify(nngxx::pipe_ev::rem_post, pipe_cb, this).or_abort();
203 socket_m.pipe_notify(nngxx::pipe_ev::num, pipe_cb, this).or_abort();
204 }
