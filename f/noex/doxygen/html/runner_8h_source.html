54 void start_thread(std::size_t spec_hash,
55 std::packaged_task<void(std::stop_token, job)> task, job j)
56 {
57 auto j_id = j.id();
58
59 auto s_id = j.socket_id();

62
63 futures_m.emplace_back(j_id, s_id, spec_hash, task.get_future());

65 auto th_res = threads_m.try_emplace(j_id, std::move(task), std::move(j));

101 void exec(job j)
102 {
103 auto spec_hash = j.spec_hash();
104
105 auto s_id = j.socket_id();
106
107 auto lock = hf_registry_m.lock();
108
109 if (!hf_registry_m.has_handler_for(s_id, spec_hash))
110 {
111 pars::err(SL, lf::event,
112 "Unable to find handler for Spec 0x{:X} on Socket {}, skip "
113 "message ...",
114 spec_hash, s_id);
115
116 return;
117 }
118
119 auto& hf = hf_registry_m.handler_for(s_id, spec_hash);
120
121 lock.unlock();
122
123 pars::debug(SL, lf::event, "Job #{}: Running Handler [{}]", j.id(),
124 demangle(hf_registry_m.type_for(spec_hash)->name()));
125
126 try
127 {
128 {
129 auto guard = std::lock_guard{mtx_m};
130
131 process_futures();
132 }
133
134 hf(std::move(j));
135 }
136 catch (...)
137 {
138 process_exception(s_id, spec_hash);
139 }

159 void remove_pipe(const net::pipe& p)
160 {
161 auto guard = std::lock_guard{mtx_m};
162
163 auto p_id = p.id();
164
165 if (!pipe_jobs_m.at(p_id).empty())

174 void associate_job_to_pipe(const int j_id, const int p_id)
175 {
176 if (j_id <= 0)
177 throw std::runtime_error(fmt::format("Job #{}: invalid Job!", p_id));
178
179 if (p_id <= 0)
180 return;
181
182 auto guard = std::lock_guard{mtx_m};
183
184 // add j_id to the jobs related to p, unless p was removed with a previous
185 // call to remove_pipe(p) (ie: pipe_jobs_m does not contains p.id())
186 if (pipe_jobs_m.contains(p_id))
187 {
188 pipe_jobs_m.at(p_id).push_back(j_id);
189
190 pars::debug(SL, lf::event,
191 "Job #{} pushed and associated to Pipe {:X} [# size: {}]",
192 j_id, p_id, pipe_jobs_m.at(p_id).size());
193 }

204private:
205 std::unordered_set<std::size_t> stop_runnings(std::vector<std::size_t> j_ids)
206 {
207 std::unordered_set<std::size_t> pending_jobs;
208
209 for (const auto& j_id : j_ids)
210 {
211 if (thread_running(j_id))
212 {
213 // stop running job, if possible
214 if (stop_possible(j_id))
215 request_stop(j_id);
216 }
217 else
218 {
219 pending_jobs.insert(j_id);
220 }
221 }
222
223 return pending_jobs;
224 }

226 void process_futures()
227 {
228 for (auto it = futures_m.begin(); it != futures_m.end();)
229 {
230 using namespace std::chrono_literals;
231
232 auto& [j_id, s_id, spec_hash, f] = *it;
233
234 if (f.wait_for(0ms) == std::future_status::ready)
235 {
236 try
237 {
238 f.get();
239 }
240 catch (...)
241 {
242 pars::debug(SL, lf::event,
243 "Job #{}: Throws, processing exceptions !!", j_id);
244
245 process_exception(s_id, spec_hash);
246 }
247
248 threads_m.erase(j_id);
249
250 pars::debug(SL, lf::event, "Job #{}: Done! [# futures: {}]", j_id,
251 futures_m.size());
252
253 it = futures_m.erase(it);
254 }
255 else
256 {
257 ++it;
258 }
259 }
260 }

262 void process_exception(auto s_id, auto spec_hash)
263 {
264 auto e_ptr = std::current_exception();
265
266 try
267 {
268 std::rethrow_exception(e_ptr);
269 }
270 catch (std::exception& e)
271 {
272 pars::err(SL, lf::event, "Handler for 0x{:X} throws: {}", spec_hash,
273 e.what());
274 }
275 catch (...)
276 {
277 pars::err(SL, lf::event, "Handler for 0x{:X} throws: Unknown Exception",
278 spec_hash);
279 }
280
281 auto e_hash = spec<fired<exception>>::hash;
282
283 auto lock = hf_registry_m.lock();
284
285 if (!hf_registry_m.has_handler_for(s_id, e_hash))
286 return;
287
288 auto& hf = hf_registry_m.handler_for(s_id, e_hash);
289
290 lock.unlock();
291
292 try
293 {
294 hf(make_job(next_job_id(), fired{exception{e_ptr}, {}}));
295 }
296 catch (std::exception& e)
297 {
298 pars::err(SL, lf::event, "Exception Handler for 0x{:X} throws: {}",
299 spec_hash, e.what());
300 }
301 catch (...)
302 {
303 pars::err(SL, lf::event,
304 "Exeption Handler for 0x{:X} throws: Unknown Exception",
305 spec_hash);
306 }
307 }
