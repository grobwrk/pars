82template<typename state_t>
83struct state_machine
84{
85 using state_type = state_t;
86
87 state_machine(state_type s)
88 : state_m{s}
89 {
90 }
91
92 void ensure(std::initializer_list<state_type> expected)
93 {
94 for (const auto& e : expected)
95 if (e == state_m)
96 return;
97
98 std::string states_str;
99
100 for (const auto& e : expected)
101 if (e != *expected.end())
102 states_str += fmt::format("{},", e);
103 else
104 states_str += fmt::format("{}", e);
105
106 throw std::runtime_error(
107 fmt::format("Wrong State {} [must be in \"{}\"]", state_m,
108 std::string_view{states_str.begin(), states_str.end() - 1}));
109 }
110
111 void ensure(state_type s)
112 {
113 if (state_m != s)
114 throw std::runtime_error(
115 fmt::format("Wrong State \"{}\" [must be \"{}\"]", state_m, s));
116 }
117
118 void next(state_type s)
119 {
120 if (next_m)
121 throw std::runtime_error("Another transition in progress!");
122
123 pars::debug(SL, lf::app, "Start transition from \"{}\" to \"{}\"", state_m,
124 s);
125
126 next_m = s;
127 }
128
129 void rollback()
130 {
131 if (!next_m)
132 throw std::runtime_error("Rollback while no transition!");
133
134 pars::debug(SL, lf::app, "Rollback transition from \"{}\" to \"{}\"",
135 state_m, *next_m);
136
137 next_m.reset();
138 }
139
140 void commit()
141 {
142 if (!next_m)
143 throw std::runtime_error("Commit while no transition!");
144
145 pars::debug(SL, lf::app, "Commit transition from \"{}\" to \"{}\"", state_m,
146 *next_m);
147
148 state_m = *next_m;
149
150 next_m.reset();
151 }
152
153 state_tx<state_type> tx(state_type to_state)
154 {
155 return state_tx(*this, to_state);
156 }
157
158 state_tx<state_type> tx(state_type from_state, state_type to_state)
159 {
160 ensure(from_state);
161
162 return state_tx(*this, to_state);
163 }
164
165 state_tx<state_type> tx(std::initializer_list<state_type> from_states,
166 state_type to_state)
167 {
168 ensure(from_states);
169
170 return state_tx(*this, to_state);
171 }
172
173 state_type current() const { return state_m; }
174
175 void reset()
176 {
177 pars::debug(SL, lf::app, "Reset State to \"{}\" [was \"{}\"]",
178 state_type::INITIALIZING, state_m);
179
180 state_m = state_type::INITIALIZING;
181 }
182
183private:
184 state_type state_m;
185 std::optional<state_type> next_m;
186};
